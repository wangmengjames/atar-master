\textbf{Question 2}

$x=6$

\textit{Marking guide:}
\begin{itemize}[nosep]
  \item Combine LHS: $\log_e(\frac{x^2}{x+2})$
  \item Equate arguments: $x^2 = 3(x+2)$
  \item Solve quadratic $x^2 - 3x - 6 = 0$, reject negative solution.
\end{itemize}

\textbf{Question 3}

$2x\sin(2x) + 2x^2\cos(2x)$

\textit{Marking guide:}
\begin{itemize}[nosep]
  \item $u=x^2, v=\sin(2x)$
  \item $u'=2x, v'=2\cos(2x)$
  \item Apply $u'v + uv'$
\end{itemize}

\textbf{Question 4}

$e^2 - 1$

\textit{Marking guide:}
\begin{itemize}[nosep]
  \item Integral $\int_0^2 e^x dx$
  \item $[e^x]_0^2 = e^2 - e^0$
\end{itemize}

\textbf{Question 5}

$n=20, p=0.6$

\textit{Marking guide:}
\begin{itemize}[nosep]
  \item $np = 12$
  \item $np(1-p) = 4.8 \implies 12(1-p) = 4.8$
  \item $1-p = 0.4 \implies p=0.6$
  \item $n(0.6)=12 \implies n=20$
\end{itemize}
