\textbf{Q1}: (A)

Area $= 5 \times 2 = 10$.

\textbf{Q2}: (A)

Area $= \frac{1}{2}(4)(4) = 8$.

\textbf{Q3}: (A)

$\left[\frac{x^3}{3}\right]_0^2 = \frac{8}{3}$.

\textbf{Q4}: (A)

$\int_a^b f(x) \, dx$.

\textbf{Q5}: (A)

$\left[\frac{3x^2}{2}\right]_0^2 = 6$.

\textbf{Q6}: (A)

$\left[\frac{x^2}{2} + x\right]_0^3 = \frac{9}{2} + 3 = \frac{15}{2}$.

\textbf{Q7}: (A)

$\frac{1}{3}$.

\textbf{Q8}: (A)

Area $= 5 \times 4 = 20$.

\textbf{Q9}: (A)

$[x^2]_1^3 = 9 - 1 = 8$.

\textbf{Q10}: (A)

$\frac{16}{4} = 4$.

\textbf{Q11}: (A)

$\left[\frac{x^3}{3}\right]_0^3 = 9$.

\textbf{Q12}: (A)

$\left[4x - \frac{x^2}{2}\right]_0^4 = 16 - 8 = 8$.

\textbf{Q13}: (A)

$\left[\frac{x^2}{2}\right]_0^3 = \frac{9}{2}$.

\textbf{Q14}: (A)

$-\left[\frac{x^3}{3} - 4x\right]_{-2}^2 = -\left[(\frac{8}{3}-8) - (-\frac{8}{3}+8)\right] = -\left[-\frac{16}{3} - \frac{16}{3}\right] = \frac{32}{3}$.

\textbf{Q15}: (A)

$\left[\frac{x^2}{2} - \frac{x^3}{3}\right]_0^1 = \frac{1}{2} - \frac{1}{3} = \frac{1}{6}$.

\textbf{Q16}: (A)

$x(x - 1) = 0$, so $x = 0$ and $x = 1$.

\textbf{Q17}: (A)

$\int_{-2}^0 |x|\,dx + \int_0^3 x\,dx = 2 + \frac{9}{2} = \frac{13}{2}$.

\textbf{Q18}: (A)

$\left[\frac{x^3}{3} + x\right]_0^2 = \frac{8}{3} + 2 = \frac{14}{3}$.

\textbf{Q19}: (A)

$\left[4x - \frac{x^3}{3}\right]_{-2}^2 = (8 - \frac{8}{3}) - (-8 + \frac{8}{3}) = \frac{32}{3}$.

\textbf{Q20}: (A)

$\frac{2(27)}{3} = 18$.

\textbf{Q21}: (A)

$2x(x-1) = 0$: $x = 0, 1$. Area $= \int_0^1 (2x - 2x^2)\,dx = [x^2 - \frac{2x^3}{3}]_0^1 = 1 - \frac{2}{3} = \frac{1}{3}$.

\textbf{Q22}: (A)

$\int_{-\sqrt{2}}^{\sqrt{2}} (4 - 2x^2)\,dx = 2\int_0^{\sqrt{2}} (4 - 2x^2)\,dx = 2[4x - \frac{2x^3}{3}]_0^{\sqrt{2}} = 2(4\sqrt{2} - \frac{4\sqrt{2}}{3}) = \frac{16\sqrt{2}}{3}$.

\textbf{Q23}: (A)

$[e^x]_0^2 = e^2 - 1$.

\textbf{Q24}: (A)

By symmetry: $2\int_0^1 |x^3 - x|\,dx = 2\int_0^1 (x - x^3)\,dx = 2[\frac{x^2}{2} - \frac{x^4}{4}]_0^1 = 2 \cdot \frac{1}{4} = \frac{1}{2}$.

\textbf{Q25}: (A)

$\int_{-1}^2 (x + 2 - x^2)\,dx = [\frac{x^2}{2} + 2x - \frac{x^3}{3}]_{-1}^2 = (2 + 4 - \frac{8}{3}) - (\frac{1}{2} - 2 + \frac{1}{3}) = \frac{9}{2}$.

\textbf{Q26}: (A)

Area $= |{-2}| = 2$.

\textbf{Q27}: (A)

$[-\cos(x)]_0^{\pi} = 1 + 1 = 2$.

\textbf{Q28}: (A)

$\int_0^2 (4 - x^2)\,dx = [4x - \frac{x^3}{3}]_0^2 = 8 - \frac{8}{3} = \frac{16}{3}$.

\textbf{Q29}: (A)

$\int_0^2 (2x - x^2)\,dx = [x^2 - \frac{x^3}{3}]_0^2 = 4 - \frac{8}{3} = \frac{4}{3}$.

\textbf{Q30}: (A)

$[\ln(x)]_1^{e^2} = 2 - 0 = 2$.

\textbf{Q31}: (A)

$[e^x - x]_0^1 = (e - 1) - (1 - 0) = e - 2$.

\textbf{Q32}: (A)

$\int_0^{\pi/4} (\cos x - \sin x)\,dx = [\sin x + \cos x]_0^{\pi/4} = \sqrt{2} - 1$.

\textbf{Q33}: (A)

$\int_0^1 (e^x - e^{-x})\,dx = [e^x + e^{-x}]_0^1 = (e + e^{-1}) - 2 = e + \frac{1}{e} - 2$.

\textbf{Q34}: (A)

$\int_0^1 (1 - x^2)\,dx + \int_1^2 (x^2 - 1)\,dx = \frac{2}{3} + \frac{4}{3} = 2$.
